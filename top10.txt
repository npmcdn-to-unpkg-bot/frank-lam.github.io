SELECT   TOP (15) USERINFO.NAME, CHECKINOUT.CHECKTIME
FROM      CHECKINOUT INNER JOIN
                USERINFO ON CHECKINOUT.USERID = USERINFO.USERID
WHERE   (DATEDIFF(dd, CHECKINOUT.CHECKTIME, GETDATE()) = 0)
ORDER BY CHECKINOUT.CHECKTIME



SELECT   COUNT(DISTINCT USERINFO.NAME) AS total
FROM      CHECKINOUT INNER JOIN
                USERINFO ON CHECKINOUT.USERID = USERINFO.USERID
WHERE   (DATEDIFF(dd, CHECKINOUT.CHECKTIME, GETDATE()) = 0)
